处理流程 (伪代码)
================================================================================

FOR page = 1 TO 27:
  articles = posts.list(page, per_page=100)
  
  FOR each article IN articles:
    # 步骤 1: 检查是否已优化
    IF article.meta.jetpack_seo_html_title EXISTS:
      SKIP  # 已有 SEO 标题，跳过
    ELSE:
      # 步骤 2: 生成 SEO 内容
      seo = claude_api.generate({
        title: article.title,
        excerpt: article.excerpt,
        constraints: {
          title_length: "45-60",
          description_length: "120-160"
        }
      })
      
      # 步骤 3: 更新到 WordPress
      IF seo.optimizedTitle AND seo.metaDescription:
        posts.update(article.id, {
          meta: {
            jetpack_seo_html_title: seo.optimizedTitle,
            advanced_seo_description: seo.metaDescription
          }
        })
        success_count++
      ELSE:
        failed_ids.append(article.id)
        failed_count++
    
    # 步骤 4: 报告进度
    IF (processed % 10 == 0):
      PRINT "[{processed}/2645] 批次 X: Y/10 成功"
    
    IF (processed IN [100, 500, 1000, 1500, 2000, 2500, 2645]):
      PRINT milestone_report()
    
    time.sleep(0.5)  # 速率限制
